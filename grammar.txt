<RESULT>  ::= <Chain>

<Chain>   ::= (<Statement> ";")*
<Statement> ::= <STDfunc> | <Assign>

<Assign>  ::= <Var> "=" <Expr>
<STDfunc> ::= <Input> | <Output>

<Input>   ::= "in"  "(" <Var> ")"
<Output>  ::= "out" "(" <Expr> ")"

<Expr>    ::= <MulDiv>  (("+" | "-") <MulDiv>)*
<MulDiv>  ::= <Power> (("*" | "/") <Power>)*

<Power>   ::= <Primary> ("^" <Power>)?
<Primary> ::= "(" <Expr> ")" | <Func> | <Var> | <Number>

<Func>	  ::= <ID> "(" <Expr> ")"
<Var>	  ::= <ID>

<ID>	  ::= ([a-z] | "_") ([a-z] | [0-9] | "_")*
<Number>  ::= [0-9]+
